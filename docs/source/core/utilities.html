    convertTime converts a time value into its component string suffix and (millisecond) number value properties; returns an array

    Examples:

    convertTime(5000);
-> ['ms', 5000]

convertTime('50%');
-> ['%', 50]

convertTime('5000ms');
-> ['ms', 5000]

convertTime('5s');
-> ['ms', 5000]

    export const convertTime = (item) => {

    if (isa_number(item)) return [MS, item];

    if (item.substring) {

        const a = item.match(/^\d+\.?\d*(\D*)/);
        let timeUnit = (a[1].toLowerCase) ? a[1].toLowerCase() : MS;

        let timeValue = parseFloat(item);

        if (!isNaN(timeValue)) {

            switch (timeUnit) {

                case 's':
                    timeValue *= 1000;
                    break;

                case '%':
                    break;

                default:
                    timeUnit = MS;
            }

            return [timeUnit, timeValue];
        }
        return [MS, 0];
    }
};
