    scrawl.addNativeListener(
    ['input', 'change'], 
    (e) => {

        let a;

        if (e.target.id === 'opacity') a = e.target.value;
        else {

            let temp = e.target.value.split(',');
            a = [];

            temp.forEach(t => {
                let n = parseInt(t, 10);
                if (n.toFixed && !isNaN(n)) a.push(n)
            });
        }

        myFilter.set({ 
            [e.target.id]: a,
        });
    }, 
    '.controlItem');
