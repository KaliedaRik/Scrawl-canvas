    const stackCheck = (function () {

    let stackActive = false;
    const here = stack.here;

    return function () {

        if (here.active != stackActive) {

            stackActive = here.active;

            pin.set({
                lockTo: (stackActive) ? 'mouse' : 'start',
            });
        }
    }
})();
